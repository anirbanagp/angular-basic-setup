import {
  suburbList,
  stateList,
  updateContactInfo
} from "./../../shared/constant";
import { BaseService } from "./../../services/base.service";
import { BaseComponent } from "./../../base.componant";
import { Component, OnInit } from "@angular/core";
import { usersContactInfo } from "../../shared/constant";
import {
  StateListOption,
  SuburbListOption,
  ContactInfoData
} from "./../../shared/models/contact-info";

@Component({
  selector: "app-contact-info",
  templateUrl: "./contact-info.component.html",
  styles: []
})
export class ContactInfoComponent extends BaseComponent implements OnInit {
  stateListOption_1: Array<StateListOption>;
  suburbListOption_1: Array<SuburbListOption>;
  stateListOption_2: Array<StateListOption>;
  suburbListOption_2: Array<SuburbListOption>;
  selectedSuburbField: string;
  show2ndSlot: boolean = false;

  contactInfoData: Array<ContactInfoData> = [];

  constructor(public baseService: BaseService) {
    super(baseService);
    this.contactInfoData[0] = new ContactInfoData();
    this.baseService.baseApi.getUsersContactInfo();
    this.baseService.baseApi.getStateList();
  }

  ngOnInit() {}

  handleApiResponse(data: any) {
    if (data.resulttype === usersContactInfo) {
      this.contactInfoData = this.baseService.global.getContactInfoData(
        data.result.result
      );
      if (this.contactInfoData[0] && this.contactInfoData[0].ks_state_id) {
        this.stateChangeHandler(
          this.contactInfoData[0].ks_state_id,
          this.contactInfoData[0].ks_suburb_id
        );
      }
    }

    if (data.resulttype === suburbList) {
      this[this.selectedSuburbField] = data.result.result;
    }

    if (data.resulttype === stateList) {
      this.stateListOption_1 = data.result.result;
      this.stateListOption_2 = data.result.result;
    }

    if (data.resulttype === updateContactInfo) {
      if (data.result.status === 200) {
        this.baseService.showMessage("successfully updated!");
      }
    }
  }

  stateChangeHandler(stateId: any, suburbField: string) {
    this.baseService.baseApi.getSuburbsList(stateId);
    this.selectedSuburbField = suburbField;
  }
  suburbChangeHandler(
    suburbValue: any,
    respectivesuburbList: any,
    postalCodeField: string
  ) {
    const selectedSuburb = respectivesuburbList.filter(
      each => each.ks_suburb_id === suburbValue
    );
    this.contactInfoData[postalCodeField] = selectedSuburb
      ? selectedSuburb[0].suburb_postcode
      : "";
  }
  toggle2ndSlot(state: boolean) {
    if (state === true) {
      if (!this.contactInfoData[1]) {
        this.contactInfoData[1] = new ContactInfoData();
      } else {
        if (this.contactInfoData[1].ks_state_id) {
          this.stateChangeHandler(
            this.contactInfoData[1].ks_state_id,
            this.contactInfoData[1].ks_suburb_id
          );
        }
      }
    } else {
      this.contactInfoData.splice(1, 1);
    }
    this.show2ndSlot = state;
  }
  updateProfile() {
    let updateData = { token: "", data: [] };
    for (let eachSet of this.contactInfoData) {
      if (eachSet) {
        updateData.data.push(eachSet);
      }
    }
    this.baseService.baseApi.updateContactInfo(updateData);
  }
}
